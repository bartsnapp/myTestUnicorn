\begin{theorem}[Intermediate Value Theorem]
If $f$ is a continuous function for all $x$ in the closed interval
$[a,b]$ and $d$ is between
\wordChoice{\choice{$a$}\choice[correct]{$f(a)$}} and
\wordChoice{\choice{b}\choice[correct]{$f(b)$}}, then there is a
number \wordChoice{\choice{$c$}\choice{$f(c)$}} in
  \wordChoice{\choice{$(a,b)$}\choice[correct]{$[a, b]$}} such that
  \wordChoice{\choice[correct]{$f(c) = d$},\choice{$f(d) = c$}}.
\end{theorem}
