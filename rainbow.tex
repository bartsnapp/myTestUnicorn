\begin{exercise}
\outcome{State the Intermediate Value Theorem including hypotheses.}
\tag{continuity}
\tag{intermediate value theorem}
\begin{theorem}[Intermediate Value Theorem]
If $f$ is a continuous function for all $x$ in the closed interval
$[b,n]$ and $u$ is between
\wordChoice{\choice{$b$}\choice[correct]{$f(b)$}} and
\wordChoice{\choice{$n$}\choice[correct]{$f(n)$}}, then there is a
number \wordChoice{\choice[correct]{$a$}\choice{$f(a)$}} in
  \wordChoice{\choice{$(b,n)$}\choice[correct]{$[b, n]$}} such that
  \wordChoice{\choice[correct]{$f(a) = u$},\choice{$f(u) = a$}}.
\end{theorem}
